   1	# DnD Weapons Phase 5 — Acquisition (Loot, Trades, Mob Drops) Implementation Plan
   2	
   3	> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
   4	
   5	**Goal:** Wire the 27 base iron-tier DnD weapons into vanilla structure loot, villager trades, and mob drops; add diamond-tier overlays for 4 endgame structures and netherite drops from Wither Skeleton / Piglin Brute / Warden; add a 100%-guaranteed random netherite weapon drop on Wither boss death.
   6	
   7	**Architecture:** Single declarative `AcquisitionCatalog` is the source of truth for all three surfaces. Thin per-concern registrars (`WeaponLootRegistrar`, `WeaponTradeRegistrar`, `WitherTrophyHandler`) read the catalog at mod init. On 26.1.2 the Fabric `TradeOfferHelper` API is absent — `Phase5TradeCodegen.kt` emits villager-trade JSON files into `data/dndweapons/villager_trades/` from the same catalog. All version drift (Fabric loot v2/v3 modules, 5/3/4-param `Modify` SAM, `location()`/`identifier()` rename, `ItemCost` vs `ItemStack` cost wrappers) is confined to the two registrar files; catalog data is version-agnostic.
   8	
   9	**Tech Stack:** Kotlin 2.3.21, Fabric Loom 1.16.2, Stonecutter 0.6, JUnit 5 for unit tests, Fabric Gametest API for runtime tests. MC target matrix: 1.20.1, 1.21.1, 1.21.4, 1.21.11, 26.1.2.

  66	 */

  70	    val tier: Tier,

  97	// src/main/kotlin/com/dndweapons/acquisition/MobDrop.kt
  98	package com.dndweapons.acquisition
  99	
 100	/**
 101	 * One Phase 5 catalog entry for a vanilla mob loot table.
 102	 *
 103	 *  - [ironWeapon] is the base weapon id dropped at iron tier (e.g. "battleaxe").
 104	 *    Null means this mob has no iron-tier drop (used for Warden, which drops a
 105	 *    random netherite weapon only).
 106	 *  - [ironPct] and [netheritePct] are pool weights for two INDEPENDENT rolls:
 107	 *    one iron pool and one netherite pool inserted into the same mob loot table.
 108	 *    Both rolls can produce a drop on the same kill (the "stacked" policy from
 109	 *    the spec, decision #6).
 110	 *  - When [ironWeapon] is null, the iron roll is skipped entirely; the netherite
 111	 *    pool emits a random selection across all 27 netherite-tier weapons.
 112	 *  - When [ironWeapon] is non-null and [netheritePct] > 0, the netherite pool
 113	 *    emits the SAME weapon at netherite tier (e.g. Wither Skeleton drops glaive
 114	 *    at both iron and netherite tiers).
 115	 */
 116	data class MobDrop(
 117	    val ironWeapon: String?,
 118	    val ironPct: Int,
 119	    val netheritePct: Int = 0,
 120	)
 121	```

 192	```kotlin
 193	// src/main/kotlin/com/dndweapons/acquisition/WeaponLookup.kt
 194	package com.dndweapons.acquisition
 195	
 196	import com.dndweapons.DndWeaponsMod
 197	import com.dndweapons.catalog.Tier
 198	import com.dndweapons.catalog.Weapons
 199	import net.minecraft.core.registries.BuiltInRegistries
 200	import net.minecraft.world.item.Item
 201	//? if >=1.21 {
 202	import net.minecraft.resources.ResourceLocation
 203	//?} else {
 204	/*import net.minecraft.resources.ResourceLocation*/
 205	//?}
 206	
 207	/**
 208	 * Resolves (base weapon id, tier) -> registered Item by consulting the Phase 4
 209	 * `Weapons.ALL_TIERED` list and the vanilla item registry. Used by both the
 210	 * loot registrar (to convert catalog entries into LootItem stacks) and the
 211	 * trade registrar (to build MerchantOffer result stacks).
 212	 *
 213	 * Returns null when:
 214	 *  - The base id is not in `Weapons.ALL`, OR
 215	 *  - The (id, tier) pair is not in `Weapons.ALL_TIERED` (e.g. requesting diamond
 216	 *    for a vanilla-mapped weapon like shortsword, or a ranged weapon excluded
 217	 *    from the tier ladder in Phase 4).
 218	 */
 219	object WeaponLookup {
 220	
 221	    fun byId(weaponId: String, tier: Tier): Item? {
 222	        val targetId = weaponId + tier.suffix
 223	        val location = resourceLocation(DndWeaponsMod.MOD_ID, targetId) ?: return null
 224	        val item = BuiltInRegistries.ITEM.get(location)
 225	        return when {
 226	            item == null -> null
 227	            item === BuiltInRegistries.ITEM.get(resourceLocation("minecraft", "air")) -> null
 228	            else -> item
 229	        }
 230	    }
 231	
 232	    /**
 233	     * All registered netherite-tier weapons. Used by the Warden mob-drop pool
 234	     * (random selection) and the Wither trophy event handler.
 235	     */
 236	    fun allNetherite(): List<Item> = Weapons.ALL_TIERED
 237	        .filter { (_, tier) -> tier == Tier.NETHERITE }
 238	        .mapNotNull { (spec, _) -> byId(spec.id.removeSuffix("_netherite"), Tier.NETHERITE) }
 239	
 240	    private fun resourceLocation(namespace: String, path: String): ResourceLocation? {
 241	        //? if >=1.21 {
 242	        return ResourceLocation.fromNamespaceAndPath(namespace, path)
 243	        //?} else {
 244	        /*return ResourceLocation(namespace, path)*/
 245	        //?}
 246	    }
 247	}
 248	```

 423	    /**
 424	     * Init-time integrity check. Logs (does not throw) on any unknown weapon id.
 425	     * Intended for catching typos in this file; the gametest is the runtime check.
 426	     */
 427	    fun validate() {
 428	        val errors = mutableListOf<String>()
 429	        for ((tableId, loot) in STRUCTURE_LOOT) {
 430	            for (w in loot.weapons) {
 431	                if (WeaponLookup.byId(w, loot.tier) == null) {
 432	                    errors += "structure $tableId: $w (${loot.tier})"
 433	                }
 434	            }
 435	        }
 436	        for ((mobId, drop) in MOB_DROPS) {
 437	            drop.ironWeapon?.let { w ->
 438	                if (WeaponLookup.byId(w, Tier.IRON) == null) errors += "mob $mobId iron: $w"
 439	            }
 440	        }
 441	        for ((profession, levels) in VILLAGER_TRADES) {
 442	            for ((level, trades) in levels) {
 443	                for (t in trades) {
 444	                    if (WeaponLookup.byId(t.weapon, Tier.IRON) == null) {
 445	                        errors += "$profession level $level: ${t.weapon}"
 446	                    }
 447	                }
 448	            }
 449	        }
 450	        if (errors.isNotEmpty()) {
 451	            DndWeaponsMod.LOGGER.error("AcquisitionCatalog references unknown weapons: $errors")
 452	        } else {
 453	            DndWeaponsMod.LOGGER.info(
 454	                "AcquisitionCatalog validated: ${STRUCTURE_LOOT.size} structures, " +
 455	                "${MOB_DROPS.size} mobs, " +
 456	                "${VILLAGER_TRADES.values.sumOf { it.values.sumOf { l -> l.size } }} trades."
 457	            )
 458	        }
 459	    }
 460	}

 594	package com.dndweapons.loot
 595	
 596	import com.dndweapons.DndWeaponsMod
 597	import com.dndweapons.acquisition.AcquisitionCatalog
 598	import com.dndweapons.acquisition.MobDrop
 599	import com.dndweapons.acquisition.StructureLoot
 600	import com.dndweapons.acquisition.WeaponLookup
 601	import com.dndweapons.catalog.Tier
 602	//? if >=26.1.2 {
 603	/*import net.fabricmc.fabric.api.loot.v3.LootTableEvents
 604	*///?} else {
 605	import net.fabricmc.fabric.api.loot.v2.LootTableEvents
 606	//?}
 607	import net.minecraft.world.item.Item
 608	import net.minecraft.world.level.storage.loot.LootPool
 609	import net.minecraft.world.level.storage.loot.LootTable
 610	import net.minecraft.world.level.storage.loot.entries.EmptyLootItem
 611	import net.minecraft.world.level.storage.loot.entries.LootItem
 612	import net.minecraft.world.level.storage.loot.functions.SetItemCountFunction
 613	import net.minecraft.world.level.storage.loot.providers.number.ConstantValue
 614	
 615	/**
 616	 * Phase 5 loot injector. Reads [AcquisitionCatalog] and adds loot pools to the
 617	 * relevant vanilla tables. Structure chests get one pool per matching entry
 618	 * (weapon vs empty, weighted by `chancePct`). Mob entities get up to two
 619	 * INDEPENDENT pools - one iron, one netherite - that roll separately on each
 620	 * kill.
 621	 *
 622	 * Three dimensions of version drift (mirrors Phase 4's SmithingTemplateLootInjector):
 623	 *  1. Fabric module path: v2 (1.20.1..1.21.11) vs v3 (26.1.2).
 624	 *  2. Modify SAM arity:
 625	 *     - v2 / 1.20.1                : 5 params (RM, LM, ResourceLocation, Builder, Source)
 626	 *     - v2 / 1.21.1..1.21.11       : 3 params (ResourceKey<LootTable>, Builder, Source)
 627	 *     - v3 / 26.1.2                : 4 params (ResourceKey<LootTable>, Builder, Source, Registries)
 628	 *  3. ResourceKey accessor for the table id:
 629	 *     - 1.20.1                     : ResourceLocation supplied as the 3rd SAM arg directly
 630	 *     - 1.21.1, 1.21.4             : key.location().toString()
 631	 *     - 1.21.11, 26.1.2            : key.identifier().toString()
 632	 */
 633	object WeaponLootRegistrar {
 634	
 635	    private const val MOD_VERSION_STRING: String =
 636	        //? if >=1.21.11 { /*"1.21.11"*///?}
 637	        //? if (>=1.21.4) & (<1.21.11) { /*"1.21.4"*///?}
 638	        //? if (>=1.21.1) & (<1.21.4) { /*"1.21.1"*///?}
 639	        //? if <1.21.1 {
 640	        "1.20.1"
 641	        //?}
 642	
 643	    fun register() {
 644	        //? if >=26.1.2 {
 645	        /*// v3 hook on 26.1.2: SAM is (key, builder, source, registries)
 646	        LootTableEvents.MODIFY.register { key, builder, _, _ ->
 647	            injectFor(key.identifier().toString(), builder)
 648	        }
 649	        *///?}
 650	        //? if >=1.21.11 {
 651	        //? if <26.1.2 {
 652	        /*// v2 hook on 1.21.11: 3-param SAM, identifier() rename
 653	        LootTableEvents.MODIFY.register { key, builder, _ ->
 654	            injectFor(key.identifier().toString(), builder)
 655	        }
 656	        *///?}
 657	        //?}
 658	        //? if >=1.21.1 {
 659	        //? if <1.21.11 {
 660	        // v2 hook on 1.21.1, 1.21.4: 3-param SAM, location() still present
 661	        LootTableEvents.MODIFY.register { key, builder, _ ->
 662	            injectFor(key.location().toString(), builder)
 663	        }
 664	        //?}
 665	        //?}
 666	        //? if <1.21.1 {
 667	        /*// v2 hook on 1.20.1: 5-param SAM, tableId is the 3rd arg (ResourceLocation)
 668	        LootTableEvents.MODIFY.register { _, _, tableId, builder, _ ->
 669	            injectFor(tableId.toString(), builder)
 670	        }
 671	        *///?}

 675	            "${AcquisitionCatalog.MOB_DROPS.size} mob tables)."
 676	        )
 677	    }
 678	
 679	    private fun injectFor(tableId: String, builder: LootTable.Builder) {
 680	        AcquisitionCatalog.STRUCTURE_LOOT[tableId]?.let { entry ->
 681	            if (versionApplies(entry)) {
 682	                builder.withPool(buildStructurePool(entry))
 683	            }
 684	        }
 685	        AcquisitionCatalog.MOB_DROPS[tableId]?.let { drop ->
 686	            if (drop.ironPct > 0)      builder.withPool(buildMobPool(drop, Tier.IRON))
 687	            if (drop.netheritePct > 0) builder.withPool(buildMobPool(drop, Tier.NETHERITE))
 688	        }
 689	    }
 690	
 691	    private fun versionApplies(entry: StructureLoot): Boolean {
 692	        if (entry.minVersion == null) return true
 693	        // Simple lexical compare works because our version strings use the same
 694	        // segmented MAJOR.MINOR.PATCH form throughout (1.20.1 < 1.21.1 < 1.21.4 etc.).
 695	        return compareVersionStrings(MOD_VERSION_STRING, entry.minVersion) >= 0
 696	    }
 697	
 698	    /** Returns negative if a < b, 0 if equal, positive if a > b. */
 699	    private fun compareVersionStrings(a: String, b: String): Int {
 700	        val ap = a.split('.').map { it.toIntOrNull() ?: 0 }
 701	        val bp = b.split('.').map { it.toIntOrNull() ?: 0 }
 702	        for (i in 0 until maxOf(ap.size, bp.size)) {
 703	            val ai = ap.getOrNull(i) ?: 0
 704	            val bi = bp.getOrNull(i) ?: 0
 705	            if (ai != bi) return ai - bi
 706	        }
 707	        return 0
 708	    }
 709	
 710	    private fun buildStructurePool(entry: StructureLoot): LootPool.Builder {
 711	        val pool = LootPool.lootPool().setRolls(ConstantValue.exactly(1.0f))
 712	        // Weighted weapon entries: each weapon shares the chancePct weight equally.
 713	        val perWeaponWeight = entry.chancePct / entry.weapons.size
 714	        for (weaponId in entry.weapons) {
 715	            val item: Item = WeaponLookup.byId(weaponId, entry.tier) ?: continue
 716	            pool.add(
 717	                LootItem.lootTableItem(item)
 718	                    .apply(SetItemCountFunction.setCount(ConstantValue.exactly(1.0f)))
 719	                    .setWeight(maxOf(perWeaponWeight, 1))
 720	            )
 721	        }
 722	        pool.add(EmptyLootItem.emptyItem().setWeight(maxOf(100 - entry.chancePct, 1)))
 723	        return pool
 724	    }
 725	
 726	    private fun buildMobPool(drop: MobDrop, tier: Tier): LootPool.Builder {
 727	        val pool = LootPool.lootPool().setRolls(ConstantValue.exactly(1.0f))
 728	        val pct = if (tier == Tier.IRON) drop.ironPct else drop.netheritePct
 729	        if (tier == Tier.NETHERITE && drop.ironWeapon == null) {
 730	            // Warden case: random selection across all 27 netherite-tier weapons.
 731	            // Each weapon entry has weight 1; empty entry weight balances the (100-pct).
 732	            val netheriteItems = WeaponLookup.allNetherite()
 733	            val perItemWeight = maxOf(pct / netheriteItems.size, 1)
 734	            for (item in netheriteItems) {
 735	                pool.add(
 736	                    LootItem.lootTableItem(item)
 737	                        .apply(SetItemCountFunction.setCount(ConstantValue.exactly(1.0f)))
 738	                        .setWeight(perItemWeight)
 739	                )
 740	            }
 741	            pool.add(EmptyLootItem.emptyItem().setWeight(maxOf(100 - pct, 1)))
 742	        } else {
 743	            // Iron OR netherite for a mob that has a single thematic weapon id.
 744	            val weaponId = drop.ironWeapon ?: return pool   // defensive; netherite-only handled above
 745	            val item = WeaponLookup.byId(weaponId, tier) ?: return pool
 746	            pool.add(
 747	                LootItem.lootTableItem(item)
 748	                    .apply(SetItemCountFunction.setCount(ConstantValue.exactly(1.0f)))
 749	                    .setWeight(pct)
 750	            )
 751	            pool.add(EmptyLootItem.emptyItem().setWeight(maxOf(100 - pct, 1)))
 752	        }
 753	        return pool
 754	    }
 755	}

1013

1016	import java.io.File

1018	/**
1019	 * Emits the 9 villager-trade JSON files Minecraft 26.1.2 loads as a trade
1020	 * rebalance data pack. The same trades that
1021	 * [com.dndweapons.trade.WeaponTradeRegistrar] registers via Fabric API on
1022	 * 1.20.1-1.21.11 are emitted here as JSON for 26.1.2.
1023	 *
1024	 * Run via `./gradlew :26.1.2:runPhase5TradeCodegen` (or directly with `kotlin
1025	 * com.dndweapons.codegen.Phase5TradeCodegen`). The output files are committed
1026	 * to the repo; this main is only re-run when [AcquisitionCatalog.VILLAGER_TRADES]
1027	 * is edited.
1028	 */
1029	fun main() {
1030	    val outputRoot = File("src/main/resources/data/dndweapons/villager_trades")
1031	    outputRoot.mkdirs()
1032	    var written = 0
1033	    for ((professionId, levels) in AcquisitionCatalog.VILLAGER_TRADES) {
1034	        val professionSlug = professionId.removePrefix("minecraft:")
1035	        for ((level, trades) in levels) {
1036	            val fileName = "${professionSlug}_level${level}.json"
1037	            File(outputRoot, fileName).writeText(buildJson(professionId, level, trades))
1038	            written++
1039	        }
1040	    }
1041	    println("Phase5TradeCodegen: wrote $written trade JSONs to ${outputRoot.path}")
1042	}

1044	private fun buildJson(professionId: String, level: Int, trades: List<VillagerTradeEntry>): String {
1045	    val tradeEntries = trades.joinToString(",\n") { tradeJson(it) }
1046	    return """{
1047	  "profession": "$professionId",
1048	  "level": $level,
1049	  "trades": [
1050	$tradeEntries
1051	  ]
1052	}
1053	"""
1054	}
