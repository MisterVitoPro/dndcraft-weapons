   1	# DnD Weapons Phase 6 -- Wiki Generator Implementation Plan
   2	
   3	> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
   4	
   5	**Goal:** Auto-generate a player-facing GitHub Wiki from the existing `Weapons.ALL` and `AcquisitionCatalog` source-of-truth catalogs; add `generateWiki` and `publishWiki` Gradle tasks that emit to `build/wiki/` and sync to the wiki sidecar repo respectively; ship one CI workflow that publishes on `v*` / `phase-*` tag pushes.
   6	
   7	**Architecture:** A single Kotlin runnable `main` (`WikiGen.kt`) reads `Weapons.ALL` plus `AcquisitionCatalog.{STRUCTURE_LOOT, MOB_DROPS, VILLAGER_TRADES}`, builds a weapon -> acquisition-facts inverse index (`AcquisitionLookup`), renders each weapon spec through hand-rolled `buildString` templates in `WikiTemplates`, copies 11 hand-authored Markdown pages from `wiki/handwritten/` verbatim, and writes everything under `build/wiki/`. No MC API calls; no version drift -- catalog data is identical across all 5 forks. Gradle wraps the main in `generateWiki` (`JavaExec` against `:1.21.4:` runtime classpath) and `publishWiki` (shallow-clone of `<repo>.wiki.git`, copy, commit, push, auth via env var `WIKI_PUBLISH_TOKEN`).
   8	
   9	**Tech Stack:** Kotlin 2.3.21, Fabric Loom 1.16.2, Stonecutter 0.6, JUnit 5 for unit tests. No new runtime dependencies. MC target matrix unchanged: 1.20.1, 1.21.1, 1.21.4, 1.21.11, 26.1.2. The wiki generator runs against any single fork (CI pins `:1.21.4:`).

  55	
  56	```kotlin
  57	// src/main/kotlin/com/dndweapons/codegen/wiki/WikiPaths.kt
  58	package com.dndweapons.codegen.wiki
  59	
  60	import com.dndweapons.catalog.Category
  61	
  62	/**
  63	 * Filename/URL helpers for the GitHub Wiki output. GitHub Wiki uses
  64	 * hyphen-separated PascalCase filenames; spaces in display names map to hyphens.
  65	 *
  66	 * Examples:
  67	 *   "Longsword"       -> "Longsword.md"        / link target "Longsword"
  68	 *   "Hand Crossbow"   -> "Hand-Crossbow.md"    / link target "Hand-Crossbow"
  69	 *   "Light Crossbow"  -> "Light-Crossbow.md"   / link target "Light-Crossbow"
  70	 */
  71	object WikiPaths {
  72	
  73	    /** Convert a WeaponSpec.displayName to a wiki page filename (with `.md`). */
  74	    fun weaponFilename(displayName: String): String =
  75	        "${pageSlug(displayName)}.md"
  76	
  77	    /** Convert a WeaponSpec.displayName to a GitHub Wiki link target (no `.md`, hyphen-separated). */
  78	    fun pageSlug(displayName: String): String =
  79	        displayName.trim().replace(Regex("\\s+"), "-")
  80	
  81	    /** Convert a Category to its category-index filename. */
  82	    fun categoryIndexFilename(category: Category): String =
  83	        "${categoryIndexSlug(category)}.md"
  84	
  85	    /** Convert a Category to its category-index slug (link target). */
  86	    fun categoryIndexSlug(category: Category): String = when (category) {
  87	        Category.SIMPLE_MELEE   -> "Simple-Melee"
  88	        Category.SIMPLE_RANGED  -> "Simple-Ranged"
  89	        Category.MARTIAL_MELEE  -> "Martial-Melee"
  90	        Category.MARTIAL_RANGED -> "Martial-Ranged"
  91	    }
  92	
  93	    /** Human-readable label for a Category (used in headings). */
  94	    fun categoryLabel(category: Category): String = when (category) {
  95	        Category.SIMPLE_MELEE   -> "Simple Melee Weapons"
  96	        Category.SIMPLE_RANGED  -> "Simple Ranged Weapons"
  97	        Category.MARTIAL_MELEE  -> "Martial Melee Weapons"
  98	        Category.MARTIAL_RANGED -> "Martial Ranged Weapons"
  99	    }
 100	}
 101	```

 110	## Task 2: AcquisitionLookup.kt -- inverse index over AcquisitionCatalog
 111	
 112	**Files:**
 113	- Create: `src/main/kotlin/com/dndweapons/codegen/wiki/AcquisitionLookup.kt`
 114	
 115	- [ ] **Step 1: Write the lookup**
 116	
 117	```kotlin
 118	// src/main/kotlin/com/dndweapons/codegen/wiki/AcquisitionLookup.kt
 119	package com.dndweapons.codegen.wiki
 120	
 121	import com.dndweapons.acquisition.AcquisitionCatalog
 122	import com.dndweapons.catalog.Tier
 123	
 124	/**
 125	 * Inverse index: for each weapon id (e.g. "longsword"), where can the player
 126	 * acquire it? Built once at WikiGen time from the three AcquisitionCatalog maps.
 127	 *
 128	 * Used by WikiTemplates to render the "Acquisition" section of each weapon page.
 129	 */
 130	class AcquisitionLookup private constructor(
 131	    private val byWeaponId: Map<String, List<AcquisitionFact>>,
 132	) {
 133	
 134	    fun factsFor(weaponId: String): List<AcquisitionFact> =
 135	        byWeaponId[weaponId].orEmpty()
 136	
 137	    companion object {
 138	        fun build(): AcquisitionLookup {
 139	            val out = linkedMapOf<String, MutableList<AcquisitionFact>>()
 140	
 141	            // Structure loot
 142	            for ((tableId, entry) in AcquisitionCatalog.STRUCTURE_LOOT) {
 143	                val tableLabel = friendlyStructure(tableId)
 144	                for (weaponId in entry.weapons) {
 145	                    out.getOrPut(weaponId) { mutableListOf() }
 146	                        .add(
 147	                            AcquisitionFact.StructureChest(
 148	                                tableLabel = tableLabel,
 149	                                chancePct = entry.chancePct,
 150	                                tier = entry.tier,
 151	                                minVersion = entry.minVersion,
 152	                            )
 153	                        )
 154	                }
 155	            }
 156	
 157	            // Mob drops
 158	            for ((entityId, drop) in AcquisitionCatalog.MOB_DROPS) {
 159	                val mobLabel = friendlyMob(entityId)
 160	                val ironWeapon = drop.ironWeapon
 161	                if (ironWeapon != null && drop.ironPct > 0) {
 162	                    out.getOrPut(ironWeapon) { mutableListOf() }
 163	                        .add(AcquisitionFact.MobDrop(mobLabel, drop.ironPct, Tier.IRON))
 164	                }
 165	                // Netherite mob drops with a specific weapon are stacked alongside the
 166	                // iron drop; netherite-with-null-weapon (Warden) is a random pick and
 167	                // is documented on every netherite weapon page via NetheriteRandomDrop.
 168	                if (ironWeapon != null && drop.netheritePct > 0) {
 169	                    out.getOrPut(ironWeapon) { mutableListOf() }
 170	                        .add(AcquisitionFact.MobDrop(mobLabel, drop.netheritePct, Tier.NETHERITE))
 171	                }
 172	            }
 173	
 174	            // Villager trades
 175	            for ((profession, levels) in AcquisitionCatalog.VILLAGER_TRADES) {
 176	                val profLabel = friendlyProfession(profession)
 177	                for ((level, trades) in levels) {
 178	                    for (trade in trades) {
 179	                        out.getOrPut(trade.weapon) { mutableListOf() }
 180	                            .add(
 181	                                AcquisitionFact.VillagerTrade(
 182	                                    profession = profLabel,
 183	                                    level = level,
 184	                                    emeralds = trade.emeralds,
 185	                                )
 186	                            )
 187	                    }
 188	                }
 189	            }
 190	
 191	            return AcquisitionLookup(out)
 192	        }
 193	
 194	        private fun friendlyStructure(tableId: String): String = tableId
 195	            .removePrefix("minecraft:chests/")
 196	            .replace('_', ' ')
 197	            .split(' ', '/')
 198	            .joinToString(" ") { it.replaceFirstChar(Char::titlecase) }
 199	
 200	        private fun friendlyMob(entityId: String): String = entityId
 201	            .removePrefix("minecraft:entities/")
 202	            .replace('_', ' ')
 203	            .split(' ')
 204	            .joinToString(" ") { it.replaceFirstChar(Char::titlecase) }
 205	
 206	        private fun friendlyProfession(profession: String): String = profession
 207	            .removePrefix("minecraft:")
 208	            .replace('_', ' ')
 209	            .split(' ')
 210	            .joinToString(" ") { it.replaceFirstChar(Char::titlecase) }
 211	    }
 212	}
 213	
 214	/** Tagged union describing one acquisition source for one weapon. */
 215	sealed interface AcquisitionFact {
 216	    data class StructureChest(
 217	        val tableLabel: String,
 218	        val chancePct: Int,
 219	        val tier: Tier,
 220	        val minVersion: String?,
 221	    ) : AcquisitionFact
 222	
 223	    data class MobDrop(
 224	        val mobLabel: String,
 225	        val chancePct: Int,
 226	        val tier: Tier,
 227	    ) : AcquisitionFact
 228	
 229	    data class VillagerTrade(
 230	        val profession: String,
 231	        val level: Int,
 232	        val emeralds: Int,
 233	    ) : AcquisitionFact
 234	}
 235	```

 249	- [ ] **Step 1: Write the templates**
 250	
 251	```kotlin
 252	// src/main/kotlin/com/dndweapons/codegen/wiki/WikiTemplates.kt
 253	package com.dndweapons.codegen.wiki
 254	
 255	import com.dndweapons.catalog.Category
 256	import com.dndweapons.catalog.DamageType
 257	import com.dndweapons.catalog.Property
 258	import com.dndweapons.catalog.RangeKind
 259	import com.dndweapons.catalog.WeaponSpec
 260	
 261	/**
 262	 * Hand-rolled buildString-based Markdown templates. No external templating
 263	 * library. Each function returns a complete page body (no trailing newline
 264	 * normalization beyond a single trailing newline).
 265	 */
 266	object WikiTemplates {
 267	
 268	    fun renderWeaponPage(spec: WeaponSpec, lookup: AcquisitionLookup): String = buildString {
 269	        appendLine("# ${spec.displayName}")
 270	        appendLine()
 271	        appendLine(
 272	            "> **Category:** ${labelFor(spec.category)}  |  " +
 273	                "**Damage type:** ${labelFor(spec.damageType)}  |  " +
 274	                "**DnD dice:** ${spec.diceText}" +
 275	                (spec.versatileDice?.let { " (versatile $it)" } ?: "")
 276	        )
 277	        appendLine()
 278	
 279	        if (spec.isVanillaMapped) {
 280	            appendLine(renderVanillaCallout(spec))
 281	            appendLine()
 282	        }
 283	
 284	        appendLine("## Stats")
 285	        appendLine()
 286	        appendLine("| Property | Value |")
 287	        appendLine("|---|---|")
 288	        val dmgCell =
 289	            if (spec.versatileBonus > 0) "${spec.attackDamage} (+${spec.versatileBonus} versatile)"
 290	            else "${spec.attackDamage}"
 291	        appendLine("| Attack damage | $dmgCell |")
 292	        appendLine("| Attack speed | ${spec.attackSpeed} |")
 293	        appendLine("| Reach bonus | ${spec.reachBonus} |")
 294	        appendLine("| Knockback bonus | +${spec.knockbackBonus} |")
 295	        val propsCell = if (spec.properties.isEmpty()) "(none)"
 296	                        else spec.properties.sortedBy { it.name }.joinToString(", ") { labelFor(it) }
 297	        appendLine("| Properties | $propsCell |")
 298	        appendLine("| Range | ${labelFor(spec.ranged)} |")
 299	        appendLine("| Base durability | ${spec.baseDurability} |")
 300	        appendLine()
 301	
 302	        appendLine("## Tiers")
 303	        appendLine()
 304	        if (spec.isVanillaMapped) {
 305	            appendLine("Vanilla material progression applies. No smithing template required.")
 306	        } else {
 307	            appendLine(
 308	                "This weapon supports the full smithing-upgrade ladder " +
 309	                    "(iron -> diamond -> netherite), as described in " +
 310	                    "[Smithing-Upgrade-System](Smithing-Upgrade-System)."
 311	            )
 312	        }
 313	        appendLine()
 314	
 315	        appendLine("## Acquisition")
 316	        appendLine()
 317	        val facts = lookup.factsFor(spec.id)
 318	        if (spec.isVanillaMapped) {
 319	            appendLine(
 320	                "Acquired via the vanilla Minecraft item (see callout above). " +
 321	                    "Not present in any structure loot, villager trade, or mob drop in v1.0."
 322	            )
 323	        } else if (facts.isEmpty()) {
 324	            appendLine(
 325	                "Crafting only. Not present in any structure loot, villager trade, " +
 326	                    "or mob drop in v1.0."
 327	            )
 328	        } else {
 329	            for (fact in facts) {
 330	                appendLine("- ${renderFactLine(fact)}")
 331	            }
 332	        }
 333	        appendLine()
 334	
 335	        appendLine("## Combat behaviour")
 336	        appendLine()
 337	        appendLine(
 338	            "Properties translated to vanilla-feeling hooks per the " +
 339	                "[Combat-Mechanics](Combat-Mechanics) page."
 340	        )
 341	        if (spec.properties.isNotEmpty()) {
 342	            appendLine()
 343	            for (prop in spec.properties.sortedBy { it.name }) {
 344	                appendLine("- **${labelFor(prop)}**: ${propertyHookSummary(prop)}")
 345	            }
 346	        }
 347	    }
 348	
 349	    fun renderCategoryIndex(category: Category, specs: List<WeaponSpec>): String = buildString {
 350	        appendLine("# ${WikiPaths.categoryLabel(category)}")
 351	        appendLine()
 352	        appendLine("| Weapon | Damage | Properties | Notes |")
 353	        appendLine("|---|---|---|---|")
 354	        for (spec in specs.filter { it.category == category }.sortedBy { it.displayName }) {
 355	            val link = WikiPaths.pageSlug(spec.displayName)
 356	            val propsCell = if (spec.properties.isEmpty()) "-"
 357	                            else spec.properties.sortedBy { it.name }.joinToString(", ") { labelFor(it) }
 358	            val notes = if (spec.isVanillaMapped) "Vanilla-mapped" else ""
 359	            appendLine("| [${spec.displayName}]($link) | ${spec.attackDamage} | $propsCell | $notes |")
 360	        }
 361	    }
 362	
 363	    fun renderHomeHeader(modVersion: String, buildSha: String): String = buildString {
 364	        appendLine("<!-- AUTO-GENERATED HEADER -->")
 365	        appendLine(
 366	            "**DnD Weapons** for Minecraft -- v$modVersion, build $buildSha. " +
 367	                "Supported MC versions: 1.20.1, 1.21.1, 1.21.4, 1.21.11, 26.1.2."
 368	        )
 369	        appendLine()
 370	        appendLine("----")
 371	        appendLine()
 372	        appendLine("<!-- HANDWRITTEN BODY -->")
 373	    }
 374	
 375	    // ----- private helpers -----
 376	
 377	    private fun renderVanillaCallout(spec: WeaponSpec): String {
 378	        val vanillaName = when (spec.id) {
 379	            "shortsword"     -> "Iron Sword (and any vanilla sword tier)"
 380	            "shortbow"       -> "Bow"
 381	            "light_crossbow" -> "Crossbow"
 382	            "trident"        -> "Trident"
 383	            else             -> "vanilla item"
 384	        }
 385	        return ":information_source:  **This weapon is represented by the vanilla Minecraft " +
 386	            "$vanillaName.** When you craft or pick up the vanilla item, it carries " +
 387	            "the ${spec.displayName} identity: tooltip stat block, combat hooks, and " +
 388	            "role tag. No separate item is registered."
 389	    }
 390	
 391	    private fun renderFactLine(fact: AcquisitionFact): String = when (fact) {
 392	        is AcquisitionFact.StructureChest -> {
 393	            val versionNote = fact.minVersion?.let { " _(${it}+ only)_" } ?: ""
 394	            "Structure chest: **${fact.tableLabel}** -- ${fact.chancePct}% (${labelFor(fact.tier)} tier)$versionNote"
 395	        }
 396	        is AcquisitionFact.MobDrop ->
 397	            "Mob drop: **${fact.mobLabel}** -- ${fact.chancePct}% (${labelFor(fact.tier)} tier)"
 398	        is AcquisitionFact.VillagerTrade ->
 399	            "Villager trade: **${fact.profession}** level ${fact.level} -- ${fact.emeralds} emeralds"
 400	    }
 401	
 402	    private fun propertyHookSummary(prop: Property): String = when (prop) {
 403	        Property.LIGHT       -> "+1 damage when offhand also holds a Light weapon (dual-wield)"
 404	        Property.HEAVY       -> "+1 knockback level on hit"
 405	        Property.FINESSE     -> "+20% damage when attacker is sprinting"
 406	        Property.VERSATILE   -> "+versatile damage bonus when wielded two-handed"
 407	        Property.TWO_HANDED  -> "Requires both hands; offhand items prevent attack"
 408	        Property.REACH       -> "+1 block attack range"
 409	        Property.THROWN      -> "Right-click to throw as a ranged projectile"
 410	        Property.AMMUNITION  -> "Requires the appropriate ammo item"
 411	        Property.LOADING     -> "Reload animation between shots"
 412	        Property.SPECIAL     -> "See weapon-specific notes"
 413	    }
 414	
 415	    private fun labelFor(category: Category): String = WikiPaths.categoryLabel(category)
 416	        .removeSuffix(" Weapons")
 417	
 418	    private fun labelFor(dt: DamageType): String = when (dt) {
 419	        DamageType.SLASHING    -> "Slashing"
 420	        DamageType.PIERCING    -> "Piercing"
 421	        DamageType.BLUDGEONING -> "Bludgeoning"
 422	    }
 423	
 424	    private fun labelFor(p: Property): String =
 425	        p.name.lowercase().replaceFirstChar(Char::titlecase).replace('_', ' ')
 426	
 427	    private fun labelFor(r: RangeKind): String = when (r) {
 428	        RangeKind.NONE   -> "Melee"
 429	        RangeKind.THROWN -> "Thrown"
 430	        RangeKind.BOW    -> "Bow"
 431	        RangeKind.CROSSBOW -> "Crossbow"
 432	        RangeKind.FIREARM  -> "Firearm"
 433	    }
 434	
 435	    private fun labelFor(t: com.dndweapons.catalog.Tier): String = when (t) {
 436	        com.dndweapons.catalog.Tier.IRON      -> "Iron"
 437	        com.dndweapons.catalog.Tier.DIAMOND   -> "Diamond"
 438	        com.dndweapons.catalog.Tier.NETHERITE -> "Netherite"
 439	    }
 440	}

 452	## Task 4: WikiGen.kt -- runnable main
 453	
 454	**Files:**
 455	- Create: `src/main/kotlin/com/dndweapons/codegen/WikiGen.kt`
 456	
 457	- [ ] **Step 1: Write the runnable**
 458	
 459	```kotlin
 460	// src/main/kotlin/com/dndweapons/codegen/WikiGen.kt
 461	package com.dndweapons.codegen
 462	
 463	import com.dndweapons.catalog.Category
 464	import com.dndweapons.catalog.Weapons
 465	import com.dndweapons.codegen.wiki.AcquisitionLookup
 466	import com.dndweapons.codegen.wiki.WikiPaths
 467	import com.dndweapons.codegen.wiki.WikiTemplates
 468	import java.nio.file.Files
 469	import java.nio.file.Path
 470	import java.nio.file.StandardCopyOption
 471	import kotlin.io.path.createDirectories
 472	import kotlin.io.path.exists
 473	import kotlin.io.path.readText
 474	import kotlin.io.path.writeText
 475	import kotlin.streams.toList
 476	
 477	/**
 478	 * Phase 6 wiki generator. Reads Weapons.ALL and AcquisitionCatalog from the
 479	 * compiled source set and emits Markdown under args[0] (default: build/wiki/).
 480	 *
 481	 * Run via Gradle:
 482	 *   ./gradlew :1.21.4:generateWiki
 483	 *
 484	 * Direct invocation:
 485	 *   java -cp <main runtime classpath> com.dndweapons.codegen.WikiGenKt <out-dir>
 486	 */
 487	object WikiGen {
 488	
 489	    fun run(outDir: Path, modVersion: String, buildSha: String, handwrittenDir: Path) {
 490	        val weaponsOut = outDir.resolve("Weapons")
 491	        weaponsOut.createDirectories()
 492	
 493	        val lookup = AcquisitionLookup.build()
 494	        val allSpecs = Weapons.ALL
 495	
 496	        // 1. Per-weapon pages
 497	        for (spec in allSpecs) {
 498	            val md = WikiTemplates.renderWeaponPage(spec, lookup)
 499	            weaponsOut.resolve(WikiPaths.weaponFilename(spec.displayName)).writeText(md)
 500	        }
 501	
 502	        // 2. Category indexes
 503	        for (cat in Category.values()) {
 504	            val md = WikiTemplates.renderCategoryIndex(cat, allSpecs)
 505	            weaponsOut.resolve(WikiPaths.categoryIndexFilename(cat)).writeText(md)
 506	        }
 507	
 508	        // 3. Handwritten pages (verbatim) + Home.md (mixed)
 509	        if (!handwrittenDir.exists()) {
 510	            error("Handwritten dir does not exist: $handwrittenDir")
 511	        }
 512	        Files.list(handwrittenDir).use { stream ->
 513	            for (path in stream.toList()) {
 514	                val name = path.fileName.toString()
 515	                if (!name.endsWith(".md")) continue
 516	                val target = outDir.resolve(name)
 517	                if (name == "Home.md") {
 518	                    val header = WikiTemplates.renderHomeHeader(modVersion, buildSha)
 519	                    target.writeText(header + "\n" + path.readText())
 520	                } else {
 521	                    Files.copy(path, target, StandardCopyOption.REPLACE_EXISTING)
 522	                }
 523	            }
 524	        }
 525	
 526	        // 4. Summary
 527	        val pageCount = allSpecs.size + Category.values().size
 528	        val handwrittenCount = Files.list(handwrittenDir).use { it.count() }
 529	        println("WikiGen: wrote $pageCount auto-generated pages and $handwrittenCount handwritten pages into $outDir")
 530	    }
 531	
 532	    @JvmStatic

 848	
 849	Append the following block to `build.gradle.kts`, after the existing `tasks.withType<KotlinCompile>` blocks (or at the bottom of the file if those are not present). The task block is identical across all Stonecutter forks -- no version drift, no stonecutter directives needed.
 850	
 851	```kotlin
 852	// ===== Phase 6: Wiki generator =====
 853	
 854	tasks.register<JavaExec>("generateWiki") {
 855	    group = "wiki"
 856	    description = "Generate the player-facing Markdown wiki under build/wiki/."
 857	    classpath = sourceSets["main"].runtimeClasspath
 858	    mainClass.set("com.dndweapons.codegen.WikiGen")
 859	    val out = layout.buildDirectory.dir("wiki").get().asFile
 860	    val hand = rootProject.layout.projectDirectory.dir("wiki/handwritten").asFile
 861	    val sha = try {
 862	        providers.exec {
 863	            commandLine("git", "rev-parse", "--short", "HEAD")
 864	        }.standardOutput.asText.get().trim()
 865	    } catch (_: Exception) {
 866	        "local"
 867	    }
 868	    args(out.absolutePath, modVersion, sha, hand.absolutePath)
 869	    dependsOn("classes")
 870	    doFirst {
 871	        out.mkdirs()
 872	        if (!hand.exists()) {
 873	            throw GradleException("Missing handwritten wiki dir: $hand")
 874	        }
 875	    }
 876	}
 877	
 878	tasks.register("publishWiki") {
 879	    group = "wiki"
 880	    description = "Sync build/wiki/ to <repo>.wiki.git (push). Auth via env WIKI_PUBLISH_TOKEN."
 881	    dependsOn("generateWiki")
 882	    doLast {
 883	        val wikiDir = layout.buildDirectory.dir("wiki").get().asFile
 884	        val cloneDir = layout.buildDirectory.dir("wiki-repo").get().asFile
 885	        val token: String? = System.getenv("WIKI_PUBLISH_TOKEN")
 886	        val url = if (!token.isNullOrBlank()) {
 887	            "https://x-access-token:$token@github.com/MisterVitoPro/dndcraft-weapons.wiki.git"
 888	        } else {
 889	            "git@github.com:MisterVitoPro/dndcraft-weapons.wiki.git"
 890	        }
 891	        val dryRun = project.findProperty("dryRun")?.toString().toBoolean()
 892	
 893	        logger.lifecycle("publishWiki: source = $wikiDir")
 894	        logger.lifecycle("publishWiki: target = $url")
 895	        logger.lifecycle("publishWiki: dryRun = $dryRun")
 896	
 897	        if (cloneDir.exists()) cloneDir.deleteRecursively()
 898	        cloneDir.parentFile.mkdirs()
 899	
 900	        providers.exec {
 901	            commandLine("git", "clone", "--depth", "1", url, cloneDir.absolutePath)
 902	        }.result.get()
 903	
 904	        // Wipe everything except .git
 905	        cloneDir.listFiles()?.forEach { f ->
 906	            if (f.name != ".git") {
 907	                if (f.isDirectory) f.deleteRecursively() else f.delete()
 908	            }
 909	        }
 910	
 911	        // Copy generated + handwritten wiki output into the clone
 912	        wikiDir.listFiles()?.forEach { f ->
 913	            val dest = java.io.File(cloneDir, f.name)
 914	            if (f.isDirectory) f.copyRecursively(dest, overwrite = true)
 915	            else f.copyTo(dest, overwrite = true)
 916	        }
 917	
 918	        providers.exec {
 919	            workingDir = cloneDir
 920	            commandLine("git", "add", ".")
 921	        }.result.get()
 922	
 923	        // Check whether there is anything to commit
 924	        val status = providers.exec {
 925	            workingDir = cloneDir
 926	            commandLine("git", "status", "--porcelain")
 927	        }.standardOutput.asText.get().trim()
 928	        if (status.isEmpty()) {
 929	            logger.lifecycle("publishWiki: no changes to publish, exiting clean.")
 930	            return@doLast
 931	        }
 932	
 933	        val msg = "Wiki: sync from ${project.version}"
 934	        providers.exec {
 935	            workingDir = cloneDir
 936	            commandLine("git", "-c", "user.email=wiki-bot@dndcraft-weapons", "-c", "user.name=Wiki Bot", "commit", "-m", msg)
 937	        }.result.get()
 938	
 939	        if (dryRun) {
 940	            logger.lifecycle("publishWiki: dry-run mode, NOT pushing.")
 941	            return@doLast
 942	        }
 943	        providers.exec {
 944	            workingDir = cloneDir
 945	            commandLine("git", "push", "origin", "HEAD")
 946	        }.result.get()
 947	        logger.lifecycle("publishWiki: pushed to $url")
 948	    }
 949	}
