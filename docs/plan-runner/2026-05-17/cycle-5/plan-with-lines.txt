   1	# DnD Weapons Phase 4 — Smithing Ladder Implementation Plan
   2	
   3	> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
   4	
   5	**Goal:** Add a smithing upgrade ladder (iron → diamond → netherite) to 27 registered melee + thrown DnD weapons, with two custom smithing-templates obtained via loot, wandering-trader trades, or a multi-step end-game craft.
   6	
   7	**Architecture:** Catalog-level `Tier` enum + `WeaponSpec.atTier()` extension produces 81 specs from 27 base specs; `WeaponRegistrarImpl` is parameterized on Tier and applies `.fireResistant()` for netherite. New `SmithingTemplateItems` registry covers the 2 templates + 2 cores + 4 components. Recipes/lang/models are emitted by a one-shot Kotlin codegen entry point; textures come from the `minecraft-asset-generator` agent. Phase 3 `SpecRegistry` and `PlayerAttackMixin` are unmodified — every tiered item is just another registered spec sharing the parent's properties.
   8	
   9	**Tech Stack:** Kotlin 2.3.21, Fabric Loom 1.16.2, Stonecutter 0.6, JUnit 5 for unit tests, Fabric Gametest API for runtime tests. MC target matrix: 1.20.1, 1.21.1, 1.21.4, 1.21.11, 26.1.2.
  10

 113	```kotlin
 114	// src/main/kotlin/com/dndweapons/catalog/Tier.kt
 115	package com.dndweapons.catalog
 116	
 117	/**
 118	 * Material tier of a registered DnD weapon. Iron is the baseline produced by the
 119	 * existing Phase 2b WeaponSpec catalog; Diamond and Netherite are produced by
 120	 * applying [WeaponSpec.atTier] and registered as separate items in Phase 4.
 121	 *
 122	 * Durability values mirror vanilla MC's `Tiers.IRON/DIAMOND/NETHERITE.getUses()`
 123	 * (pinned here to avoid a per-version `Tiers` class lookup). `damageBonus` is
 124	 * additive on top of the base spec's `attackDamage` and feeds directly into
 125	 * `AttributeCompat` -> `Attributes.ATTACK_DAMAGE` modifier value.
 126	 */
 127	enum class Tier(
 128	    val suffix: String,
 129	    val displayPrefix: String,
 130	    val damageBonus: Int,
 131	    val durability: Int,
 132	    val fireImmune: Boolean,
 133	) {
 134	    IRON(suffix = "", displayPrefix = "", damageBonus = 0, durability = 250, fireImmune = false),
 135	    DIAMOND(suffix = "_diamond", displayPrefix = "Diamond ", damageBonus = 1, durability = 1561, fireImmune = false),
 136	    NETHERITE(suffix = "_netherite", displayPrefix = "Netherite ", damageBonus = 2, durability = 2031, fireImmune = true),
 137	}
 138	```

 222	```kotlin
 223	/**
 224	 * Returns a copy of this WeaponSpec with the given tier applied. Tier affects:
 225	 *   - id: suffixed with "_diamond" / "_netherite" (iron leaves it untouched).
 226	 *   - displayName: prefixed with "Diamond " / "Netherite " (iron untouched).
 227	 *   - attackDamage: added to baseline (+1 diamond, +2 netherite).
 228	 *   - baseDurability: replaced with the tier's vanilla durability (250 / 1561 / 2031).
 229	 *
 230	 * Properties, attack speed, damage type, reach, and knockback carry over unchanged.
 231	 * The `vanillaRoleTag` field is preserved but in practice atTier() is only called for
 232	 * non-vanilla-mapped specs (see [Weapons.ALL_TIERED] filter).
 233	 */
 234	fun WeaponSpec.atTier(t: Tier): WeaponSpec = copy(
 235	    id = if (t == Tier.IRON) id else "$id${t.suffix}",
 236	    displayName = if (t == Tier.IRON) displayName else "${t.displayPrefix}$displayName",
 237	    attackDamage = attackDamage + t.damageBonus,
 238	    baseDurability = t.durability,
 239	)
 240	```

 832	- 54 smithing-transform recipes
 833	- 6 component crafting recipes (placeholder bodies — the real bodies are tasks 11–13)
 834	- 2 final smithing-assembly recipes (placeholder — task 14)

 838	- [ ] **Step 1: Create the codegen entry point**
 839	
 840	```kotlin
 841	// src/main/kotlin/com/dndweapons/codegen/Phase4Codegen.kt
 842	package com.dndweapons.codegen
 843	
 844	import com.dndweapons.catalog.Tier
 845	import com.dndweapons.catalog.Weapons
 846	import java.io.File
 847	
 848	/**
 849	 * One-shot codegen for Phase 4. Run from an IDE or `./gradlew :1.20.1:runMain
 850	 * -PmainClass=com.dndweapons.codegen.Phase4CodegenKt`. Writes recipes, model
 851	 * JSONs, and lang entries directly into src/main/resources/. Re-running is
 852	 * idempotent except for the lang file which is rewritten from scratch.
 853	 *
 854	 * This is NOT a gradle data-generation task; it's a manual one-time run during
 855	 * Phase 4 implementation. The resulting files are committed to the repo and
 856	 * become the source of truth thereafter.
 857	 */
 858	fun main() {
 859	    val root = File("src/main/resources")
 860	    require(root.exists()) { "Run from project root (src/main/resources must exist)" }
 861	
 862	    emitSmithingTransformRecipes(root)
 863	    emitModelJsons(root)
 864	    emitLangEntries(root)
 865	    println("Phase 4 codegen complete. 54 transform recipes, 62 models, 66 lang entries.")
 866	    println("NOTE: component/core/template-assembly recipes are NOT generated here; ")
 867	    println("      hand-author them per tasks 11-14 of the implementation plan.")
 868	}

 870	private fun emitSmithingTransformRecipes(root: File) {
 871	    val recipeDir = File(root, "data/dndweapons/recipe").also { it.mkdirs() }
 872	    for ((spec, tier) in Weapons.ALL_TIERED) {
 873	        if (tier == Tier.IRON) continue   // base iron already exists from Phase 2b crafting
 874	        val baseId = if (tier == Tier.DIAMOND) spec.id.removeSuffix("_diamond")
 875	                     else spec.id.removeSuffix("_netherite") + "_diamond"
 876	        val (templateId, additionTag) = when (tier) {
 877	            Tier.DIAMOND -> "dndweapons:diamond_weapon_upgrade_template" to "c:gems/diamond"
 878	            Tier.NETHERITE -> "dndweapons:netherite_weapon_upgrade_template" to "c:ingots/netherite"
 879	            Tier.IRON -> error("unreachable")
 880	        }
 881	        val json = """{
 882	            |  "type": "minecraft:smithing_transform",
 883	            |  "template": { "id": "$templateId" },
 884	            |  "base":     { "id": "dndweapons:$baseId" },
 885	            |  "addition": { "tag": "$additionTag" },
 886	            |  "result":   { "id": "dndweapons:${spec.id}" }
 887	            |}""".trimMargin()
 888	        File(recipeDir, "${spec.id}.json").writeText(json + "\n")
 889	    }
 890	}

 892	private fun emitModelJsons(root: File) {
 893	    val modelDir = File(root, "assets/dndweapons/models/item").also { it.mkdirs() }
 894	    val itemIds = buildList {
 895	        for ((spec, _) in Weapons.ALL_TIERED) if (spec.vanillaRoleTag == null) add(spec.id)
 896	        addAll(listOf(
 897	            "diamond_template_fragment", "weapon_smithing_binding", "diamond_template_core",
 898	            "netherite_template_fragment", "infernal_binding", "netherite_template_core",
 899	            "diamond_weapon_upgrade_template", "netherite_weapon_upgrade_template",
 900	        ))
 901	    }
 902	    for (id in itemIds.distinct()) {
 903	        val json = """{
 904	            |  "parent": "minecraft:item/generated",
 905	            |  "textures": { "layer0": "dndweapons:item/$id" }
 906	            |}""".trimMargin()
 907	        File(modelDir, "$id.json").writeText(json + "\n")
 908	    }
 909	}

 911	private fun emitLangEntries(root: File) {
 912	    val langFile = File(root, "assets/dndweapons/lang/en_us.json")
 913	    require(langFile.exists()) { "en_us.json missing — Phase 2b should have created it" }
 914	    // Read existing JSON as an ordered map by re-tokenising. We don't pull in
 915	    // a JSON library to keep codegen runnable without runtime deps.
 916	    val existing = langFile.readText()
 917	        .lines()
 918	        .filter { it.contains("\":") && !it.trim().startsWith("//") }
 919	        .associate {
 920	            val k = it.substringAfter("\"").substringBefore("\"")
 921	            val v = it.substringAfter("\": \"").substringBefore("\"")
 922	            k to v
 923	        }
 924	        .toMutableMap()
 925	
 926	    for ((spec, tier) in Weapons.ALL_TIERED) {
 927	        if (tier == Tier.IRON || spec.vanillaRoleTag != null) continue
 928	        existing["item.dndweapons.${spec.id}"] = spec.displayName
 929	    }
 930	    existing["item.dndweapons.diamond_template_fragment"] = "Diamond Template Fragment"
 931	    existing["item.dndweapons.weapon_smithing_binding"]   = "Weapon Smithing Binding"
 932	    existing["item.dndweapons.diamond_template_core"]     = "Diamond Template Core"
 933	    existing["item.dndweapons.netherite_template_fragment"] = "Netherite Template Fragment"
 934	    existing["item.dndweapons.infernal_binding"]          = "Infernal Binding"
 935	    existing["item.dndweapons.netherite_template_core"]   = "Netherite Template Core"
 936	    existing["item.dndweapons.diamond_weapon_upgrade_template"]   = "Diamond Weapon Upgrade Template"
 937	    existing["item.dndweapons.netherite_weapon_upgrade_template"] = "Netherite Weapon Upgrade Template"
 938	    existing["tooltip.dndweapons.smithing.diamond.applies_to"]    = "DnD Weapons"
 939	    existing["tooltip.dndweapons.smithing.diamond.ingredients"]   = "Diamond"
 940	    existing["tooltip.dndweapons.smithing.diamond.upgrade"]       = "Diamond Weapon Upgrade"
 941	    existing["tooltip.dndweapons.smithing.diamond.upgrade.base"]  = "DnD Weapon"
 942	    existing["tooltip.dndweapons.smithing.diamond.upgrade.additions"] = "Diamond"
 943	    existing["tooltip.dndweapons.smithing.netherite.applies_to"]  = "DnD Diamond Weapons"
 944	    existing["tooltip.dndweapons.smithing.netherite.ingredients"] = "Netherite Ingot"
 945	    existing["tooltip.dndweapons.smithing.netherite.upgrade"]     = "Netherite Weapon Upgrade"
 946	    existing["tooltip.dndweapons.smithing.netherite.upgrade.base"]  = "Diamond DnD Weapon"
 947	    existing["tooltip.dndweapons.smithing.netherite.upgrade.additions"] = "Netherite Ingot"
 948	
 949	    val sorted = existing.toSortedMap()
 950	    val sb = StringBuilder("{\n")
 951	    for ((i, e) in sorted.entries.withIndex()) {
 952	        sb.append("  \"${e.key}\": \"${e.value}\"")
 953	        if (i < sorted.size - 1) sb.append(",")
 954	        sb.append("\n")
 955	    }
 956	    sb.append("}\n")
 957	    langFile.writeText(sb.toString())
 958	}
